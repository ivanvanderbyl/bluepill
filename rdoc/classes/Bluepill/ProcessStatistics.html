    def to_s
      str = []
      @events.each do |(event, reason, time)|
        str << "  #{event} at #{time.strftime(STRFTIME)} - #{reason || "unspecified"}"
      end
      if str.size > 0
        str << "event history:"
      end
      str.reverse.join("\n")
    end
