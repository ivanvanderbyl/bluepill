      def notify(transition)
        register! if self.is_registered == false
        
        # $stdout.puts "Notifying #{self.server.host_name} of state transition #{transition.from_name} => #{transition.to_name}"
        begin
          if self.application
            Process.create(
              :application_id => self.application.id, 
              :name => self.process.name, 
              :transition => {:from_state => transition.from_name, :to_state => transition.to_name},
              :pid_file => self.process.pid_file,
              :start_command => self.process.start_command,
              :stop_command => self.process.stop_command,
              :restart_command => self.process.restart_command,
              :daemonize => self.process.daemonize,
              :working_dir => self.process.working_dir,
              :environment => self.process.environment,
              :start_grace_time => self.process.start_grace_time.to_i,
              :stop_grace_time => self.process.stop_grace_time.to_i,
              :restart_grace_time => self.process.restart_grace_time.to_i,
              :uid => self.process.uid,
              :gid => self.process.gid
            )
          end
        rescue Exception => e
          $stderr.puts e.message
          $stderr.puts e.backtrace.join("\n")
        end
      end

      def register!
        self.server = Server.create(:host_name => ::Socket.gethostname)
        self.application ||= Application.create(
          :server_id => self.server.id,
          :name => self.process.app.name
          # ,
          # :base_dir => self.process.app.base_dir,
          # :pid_file => self.process.app.pid_file,
          # :pids_dir => self.process.app.pids_dir,
          # :log_file => self.process.app.log_file,
          # :foreground => self.process.app.foreground          
        )
        $stdout.puts "Registered server #{self.server.host_name} with #{Server.site}"
        self.is_registered = true
      rescue => e
        $stderr.puts e.message
        $stderr.puts e.backtrace.join("\n")
      end
